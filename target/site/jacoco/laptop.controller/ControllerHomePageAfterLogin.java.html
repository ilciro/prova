	public static boolean logout() 
	{	
		boolean state=false;
		String n = u.getNome();
		java.util.logging.Logger.getLogger("Test logout").log(Level.INFO, "stai sloggando come {0}" ,n);
		
		if (n==null)
		{
			try {
			throw new LogoutException("Errore Logout");
			}catch(LogoutException e)
			{

			vis.setIsLogged(false);

		}

	}
